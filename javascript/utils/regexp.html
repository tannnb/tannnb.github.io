# 常用正则

# 验证相关

# 是否是金额（精确到分）

/^(0|([1-9]\d*))(\.\d{1,2})?$/

# 是否是手机号

/^1[3-9]\d{9}$/

# 是否是邮箱号

/^\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/

# 是否是 QQ 号

/^[1-9]{1}\d{4,11}$/

# 是否是链接地址

/^(https|http):\/\/[A-Za-z0-9-_]+\.[A-Za-z0-9]+[\/=\?%\-&_~`@[\]\':+!]*([^<>\"\"])*$/

# 是否是身份证号码

/(^\d{8}(0\d|11|12)([0-2]\d|30|31)\d{3}$)|(^\d{6}(18|19|20)\d{2}(0\d|11|12)([0-2]\d|30|31)\d{3}(\d|X|x)$)/

# 格式相关

# 344 格式手机号

  • 从左到右
/(^\d{3}|\d{4}\B)/g
// 例：
'15512341234'.replace(/(^\d{3}|\d{4}\B)/g,'$1 ');
  • 从右到左
/\B(?=(?:\d{4})+$)/g
// 例：
'15512341234'.replace(/\B(?=(?:\d{4})+$)/g, ' ');
